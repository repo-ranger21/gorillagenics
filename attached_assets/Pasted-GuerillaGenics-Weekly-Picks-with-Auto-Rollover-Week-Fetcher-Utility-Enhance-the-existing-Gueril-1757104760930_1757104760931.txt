// src/utils/weekFetcher.js
export async function getCurrentWeekData({ apiKey, baseUrl, staticWeek1 }) {
  const today = new Date();
  const currentDay = today.getUTCDay(); // 0=Sun, 1=Mon...
  const currentHour = today.getUTCHours();

  // Determine NFL week number (simplified: adjust based on season start date)
  const seasonStart = new Date(Date.UTC(2025, 8, 4, 0, 0, 0)); // Sept 4, 2025
  const daysSinceStart = Math.floor((today - seasonStart) / (1000 * 60 * 60 * 24));
  let weekNumber = Math.floor(daysSinceStart / 7) + 1;

  // Auto-rollover on Tuesday 12:00 AM ET
  const isTuesdayOrLater = (currentDay > 2) || (currentDay === 2 && currentHour >= 4);
  if (isTuesdayOrLater) weekNumber += 1;

  // Week 1 fallback
  if (weekNumber <= 1) return staticWeek1;

  try {
    const res = await fetch(`${baseUrl}/nfl/2025/week/${weekNumber}`, {
      headers: { 'Authorization': `Bearer ${apiKey}` }
    });
    if (!res.ok) throw new Error("API error");
    const data = await res.json();
    return data.matchups.map(m => ({
      date: m.date,
      time: m.time,
      away: m.awayTeam,
      home: m.homeTeam,
      spread: m.spread,
      total: m.overUnder,
      bioBoost: null, // placeholder until computed
      pick: null
    }));
  } catch (err) {
    console.error("Week fetch failed", err);
    return staticWeek1; // fallback to Week 1 data
  }
}
